import folium
from folium import Marker, Icon
import pandas as pd
import csv, time

df=pd.read_csv('./jongro.csv', encoding='utf-8')
df_sample=df[['수거쓰레기종류', '도로명주소', '위도', '경도']]

df_drop_sample=df_sample.dropna(axis=0)
df_final=df_drop_sample

def seoul_jongro_map(default_location=[37.57002838826, 126.97962084516], default_zoom_start=20):
  base_map=folium.Map(location=default_location, 
                      control_scale=True,
                      zoom_start=default_zoom_start)
  
  for row in df_final.itertuples():
    수거쓰레기종류, 도로명주소, 위도, 경도 = row[1:]
        
    if 수거쓰레기종류 == '의류':
      icon = Icon(color='green',icon='info-sign')
    elif 수거쓰레기종류 == '일반쓰레기':
      icon = Icon(color='blue',icon='info-sign')
    elif 수거쓰레기종류 == '재활용':
      icon = Icon(color='blue',icon='info-sign')
    elif 수거쓰레기종류 == '담배꽁초':
      icon = Icon(color='red',icon='info-sign')
    else:
      break

    Marker(location=[위도 , 경도], 
           popup=f'{수거쓰레기종류}―설치위치 : {도로명주소}',
           icon = icon).add_to(base_map)

  return base_map

print('='*100)
